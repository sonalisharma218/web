JavaScript Exception Handling Programs
1. Handle Exception for Undeclared Variable
try {
console.log(x); // x is not declared
} catch (e) {
console.log("Error caught:", e.message);
}
2. Throw "Divide by Zero" Error
function divide(a, b) {
if (b === 0) {
throw new Error("Divide by Zero");
}
return a / b;
}
try {
console.log(divide(10, 0));
} catch (e) {
console.log("Error:", e.message);
}
3. Throw "Out of Bound" Error for Array
let arr = [1, 2, 3];
try {
if (arr[5] === undefined) {
throw new Error("Out of Bound Error");
}
console.log(arr[5]);
} catch (e) {
console.log("Error:", e.message);
}
4. Exception Handling with finally Statement
try {
let result = 10 / 2;
console.log("Result:", result);
} catch (e) {
console.log("Error:", e.message);
} finally {
console.log("Finally block executed");
}
